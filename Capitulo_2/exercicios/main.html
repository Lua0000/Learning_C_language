7. Identifique os erros de compilação:

7.1

#include 

main () {
    int x, y, x;
}

R = A variável x foi declarada duas vezes.

7.2

#include 

main ()
{
    int x, y;
    float int = 5.23;
    printf("%f", int);
}

R = "int" é palavra reservada e não pode ser usada como variável. Além disso, há inconsistência de tipo.

7.3

#include 

main ()
{
    int x=y=z=0;
    printf("%d %d %d\n", x, y, z);
}

R = Declaração incorreta. O correto seria:
int x, y, z;
x = y = z = 0;

8. Possíveis causas de mau funcionamento:

8.1

#include 

main ()
{
    int n;
    scanf("Introduza um N° %d", &n);
    printf("O n° = %d\n", n);
}

R = scanf foi usado como printf. scanf serve apenas para entrada de dados.

8.2

main ()
{
    int n;
    printf("Introduza um N");
    scanf("%d\n", &n);
    printf("O n° = %d\n", n);
}

R = O uso de \n no scanf pode causar espera por mais entrada.

8.3

#include 

main()
{
    int n;
    printf("Introduza um N");
    scanf("%f", &n);
    printf("O n° = %f\n", n);
}

R = Tipo incorreto no scanf (%f para int) e erro de formatação no printf.

10. Converter data de aaaa-mm-dd para dd/mm/aaaa:

#include 

int main() {
    
    int Ano, Mes, Dia;
    
    printf("Ano: ");
    scanf("%d", &Ano);

    printf("Mes: ");
    scanf("%d", &Mes);

    printf("Dia: ");
    scanf("%d", &Dia);
    
    printf("Formato AAAA-MM-DD: %d-%d-%d\n", Ano, Mes, Dia);
    printf("Formato DD/MM/AAAA: %d/%d/%d\n", Dia, Mes, Ano);

    return 0;
}
